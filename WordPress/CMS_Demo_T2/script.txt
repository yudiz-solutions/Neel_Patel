<script src="https://code.jquery.com/jquery-3.6.0.min.js"></script>
    <script>
        // Get all the anchor tags in the sidebar
        var sidebarLinks = document.querySelectorAll('#sidebar a');

        // Loop through each anchor tag and add click event listener
        sidebarLinks.forEach(function (link) {
            link.addEventListener('click', function (e) {
                // Remove 'active' class from all anchor tags
                sidebarLinks.forEach(function (item) {
                    item.classList.remove('active');
                });

                // Add 'active' class to the clicked anchor tag
                this.classList.add('active');
            });
        });
    </script>

    SELECT pd.post_id , p1.meta_value AS p_caption , p2.meta_value AS p_hashtag FROM ( SELECT DISTINCT post_id FROM wp_meta ) AS pd LEFT JOIN wp_meta AS p1 ON p1.post_id = pd.post_id AND p1.meta_key = 'p_caption' LEFT JOIN wp_meta AS p2 ON p2.post_id = pd.post_id AND p2.meta_key = 'p_hashtag';
